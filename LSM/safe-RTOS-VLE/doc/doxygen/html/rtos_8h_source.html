796 static inline uint32_t rtos_osSuspendAllInterruptsByPriority(uint32_t suspendUpToThisPriority)
797 {
798  /* All priorities are in the range 0..15. Everything else points to an application
799  error even if the hardware wouldn't mind. */
800  assert((suspendUpToThisPriority & ~0xfu) == 0);
801 
802  /* MCU reference manual, section 28.6.6.2, p. 932: The change of the current priority
803  in the INTC should be done under global interrupt lock. */
804  rtos_osSuspendAllInterrupts();
805  const uint32_t priorityLevelSoFar = INTC.CPR_PRC0.R;
806 
807  /* It is useless and a waste of CPU but not a severe error to let this function set the
808  same priority level we already have.
809  It leads to immediate failure of the RTOS if we lower the level; however, from the
810  perspective of the application code it is not necessarily an error to do so: If an
811  application is organized in OSEK/VDX like resources it may be stringent (not
812  optimal) to acquire different resources before an operation on them is started. These
813  resources may be mapped onto different priority ceilings and the application may use
814  nested calls of GetResource to acquire all of them. We must not force the
815  application code to order the calls in order of increasing priority ceiling -- this
816  will be untransparent to the application.
817  These considerations leads to different strategies, and all of them are justified:
818  To force the application be optimal in terms of CPU consumption, we would place an
819  assertion: assert(suspendUpToThisPriority > priorityLevelSoFar);
820  To be relaxed about setting the same priority ceiling, we would place another
821  assertion: assert(suspendUpToThisPriority >= priorityLevelSoFar);
822  If we want to build an OSEK/VDX like GetResource on top of this function, we
823  should place a run-time condition: if(suspendUpToThisPriority > priorityLevelSoFar) */
824  //assert(suspendUpToThisPriority > priorityLevelSoFar);
825  //assert(suspendUpToThisPriority >= priorityLevelSoFar);
826  if(suspendUpToThisPriority > priorityLevelSoFar)
827  INTC.CPR_PRC0.R = suspendUpToThisPriority;
828 
829  /* Leave the critical section. */
830  rtos_osResumeAllInterrupts();
831 
832  /* Note, the next interrupt can still be a last one of priority less than or equal to
833  suspendUpToThisPriority. This happens occasionally when the interrupt has asserted,
834  while we were inside the critical section. Incrementing CPR does not un-assert an
835  already asserted interrupt. The isync instruction ensures that this last interrupt
836  has completed prior to the execution of the first code inside the critical section.
837  See https://community.nxp.com/message/993795 for more. */
838  asm volatile (
839 #ifdef __VLE__
840  "se_isync\n"
841 #else
842  "isync\n"
843 #endif
844  ::: "memory"
845  );
846 
847  return priorityLevelSoFar;
848 
849 } /* End of rtos_osSuspendAllInterruptsByPriority */
