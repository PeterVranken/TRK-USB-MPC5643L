672 static ALWAYS_INLINE void rtos_osSuspendAllInterrupts(void)

674  /* The completion synchronizing character of the wrteei instruction forms the memory
675  barrier, which ensures that all memory operations before the now entered critical
676  section are completed before we enter (see core RM, 4.6.1, p. 151). The "memory"
677  constraint ensures that the compiler won't reorder instructions from behind the

901 static inline bool rtos_checkUserCodeReadPtr(const void *address, size_t noBytes)
902 {
903  /* This code, like all other using core-related global linker symbols ld_*, needs
904  migration. We can't do that here and immediately as it requires a revision of the
905  linker script, which is actually useless and unwanted for the given MCU derivative.
906  We just place an assertion to avoid problem when running this code on another
907  derivative. */
909 
910  const uint8_t * const p = (uint8_t*)address;
911  extern uint8_t ld_ramStart[0], ld_ramEnd[0], ld_romStart[0], ld_romEnd[0];
912 
913  return p >= ld_ramStart && p+noBytes <= ld_ramEnd
914  || p >= ld_romStart && p+noBytes <= ld_romEnd;
915 
916 } /* End of rtos_checkUserCodeReadPtr */
