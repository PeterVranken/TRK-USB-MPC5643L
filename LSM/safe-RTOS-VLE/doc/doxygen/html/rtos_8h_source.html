623 static ALWAYS_INLINE void rtos_osSuspendAllInterrupts(void)
624 {
625  /* The completion synchronizing character of the wrteei instruction forms the memory
626  barrier, which ensures that all memory operations before the now entered critical
627  section are completed before we enter (see core RM, 4.6.1, p. 151). The "memory"
628  constraint ensures that the compiler won't reorder instructions from behind the
629  wrteei to before it. */
630  asm volatile ( /* AssemblerTemplate */
631  "wrteei 0\n"

633  : /* InputOperands */
